    public void Kilogram(String input){
        double a = Double.parseDouble(input);       
        double b = a*10;
        A2.setText(Double.toString(b)+" HG");
        double c = a*100;
        A3.setText(Double.toString(c)+" DKG");
        double d = a*1000;
        A4.setText(Double.toString(d)+" G");
        double e = a*10000;
        A5.setText(Double.toString(e)+"  DG");
        double f = a*100000;
        A6.setText(Double.toString(f)+" CG");
        double g = a*1000000;
        A7.setText(Double.toString(g)+" MG");
    }
